190     private static int _valid(String s, Type t, boolean template) {
191         boolean[] table = ENCODING_TABLES[t.ordinal()];
192 
193         for (int i = 0; i < s.length(); i++) {
194             final char c = s.charAt(i);
195             if ((c < 0x80 && c != '%' && !table[c]) || c >= 0x80) {
196                 if (!template || (c != '{' && c != '}')) {
197                     return i;
198                 }
199             }
200         }
201         return -1;
202     }

271     public static String encodeTemplateNames(String s) {
272         int i = s.indexOf('{');
273         if (i != -1)
274             s = s.replace("{", "%7B");
275         i = s.indexOf('}');
276         if (i != -1)
277             s = s.replace("}", "%7D");
278 
279         return s;
280     }

282     private static String _encode(String s, Type t, boolean template, boolean contextualEncode) {
283         final boolean[] table = ENCODING_TABLES[t.ordinal()];
284 
285         StringBuilder sb = null;
286         for (int i = 0; i < s.length(); i++) {
287             final char c = s.charAt(i);
288             if (c < 0x80 && table[c]) {
289                 if (sb != null) sb.append(c);
290             } else {
291                 if (template && (c == '{' || c == '}')) {
292                     if (sb != null) sb.append(c);
293                     continue;
294                 } else if (contextualEncode) {
295                     if (c == '%' && i + 2 < s.length()) {
296                         if (isHexCharacter(s.charAt(i + 1)) &&
297                                 isHexCharacter(s.charAt(i + 2))) {
298                             if (sb != null)
299                                 sb.append('%').append(s.charAt(i + 1)).append(s.charAt(i + 2));
300                             i += 2;
301                             continue;
302                         }
303                     }
304                 }
305 
306                 if (sb == null) {
307                     sb = new StringBuilder();
308                     sb.append(s.substring(0, i));
309                 }
310 
311                 if (c < 0x80) {
312                     if (c == ' ' && (t == Type.QUERY_PARAM)) {
313                         sb.append('+');
314                     } else {
315                         appendPercentEncodedOctet(sb, c);
316                     }
317                 } else {
318                     appendUTF8EncodedCharacter(sb, c);
319                 }
320             }
321         }
322 
323         return (sb == null) ? s : sb.toString();
324     }
325     private final static char[] HEX_DIGITS = {
326         '0', '1', '2', '3', '4', '5', '6', '7',
327         '8', '9', 'A', 'B', 'C', 'D', 'E', 'F'
328     };

330     private static void appendPercentEncodedOctet(StringBuilder sb, int b) {
331         sb.append('%');
332         sb.append(HEX_DIGITS[b >> 4]);
333         sb.append(HEX_DIGITS[b & 0x0F]);
334     }
335 
336     private static void appendUTF8EncodedCharacter(StringBuilder sb, char c) {
337         final ByteBuffer bb = UTF_8_CHARSET.encode("" + c);
338 
339         while (bb.hasRemaining()) {
340             appendPercentEncodedOctet(sb, bb.get() & 0xFF);
341         }
342     }
343     private static final String[] SCHEME = {"0-9", "A-Z", "a-z", "+", "-", "."};
344     private static final String[] UNRESERVED = {"0-9", "A-Z", "a-z", "-", ".", "_", "~"};

346     private static final boolean[][] ENCODING_TABLES = creatingEncodingTables();

348     private static boolean[][] creatingEncodingTables() {
349         boolean[][] tables = new boolean[Type.values().length][];
350 
351         List<String> l = new ArrayList<String>();
352         l.addAll(Arrays.asList(SCHEME));
353         tables[Type.SCHEME.ordinal()] = creatingEncodingTable(l);
354 
355         l.clear();
356 
357         l.addAll(Arrays.asList(UNRESERVED));
358         tables[Type.UNRESERVED.ordinal()] = creatingEncodingTable(l);
359 
360         l.addAll(Arrays.asList(SUB_DELIMS));
361 
362         tables[Type.HOST.ordinal()] = creatingEncodingTable(l);
363 
364         tables[Type.PORT.ordinal()] = creatingEncodingTable(Arrays.asList("0-9"));
365 
366         l.add(":");
367 
368         tables[Type.USER_INFO.ordinal()] = creatingEncodingTable(l);
369 
370         l.add("@");
371 
372         tables[Type.AUTHORITY.ordinal()] = creatingEncodingTable(l);
373 
374         tables[Type.PATH_SEGMENT.ordinal()] = creatingEncodingTable(l);
375         tables[Type.PATH_SEGMENT.ordinal()][';'] = false;
376 
377         tables[Type.MATRIX_PARAM.ordinal()] = tables[Type.PATH_SEGMENT.ordinal()].clone();
378         tables[Type.MATRIX_PARAM.ordinal()]['='] = false;
379 
380         l.add("/");
381 
382         tables[Type.PATH.ordinal()] = creatingEncodingTable(l);
383 
384         l.add("?");
385 
386         tables[Type.QUERY.ordinal()] = creatingEncodingTable(l);
387 
388         tables[Type.FRAGMENT.ordinal()] = tables[Type.QUERY.ordinal()];
389 
390         tables[Type.QUERY_PARAM.ordinal()] = creatingEncodingTable(l);
391         tables[Type.QUERY_PARAM.ordinal()]['='] = false;
392         tables[Type.QUERY_PARAM.ordinal()]['+'] = false;
393         tables[Type.QUERY_PARAM.ordinal()]['&'] = false;
394 
395         return tables;
396     }

398     private static boolean[] creatingEncodingTable(List<String> allowed) {
399         boolean[] table = new boolean[0x80];
400         for (String range : allowed) {
401             if (range.length() == 1) {
402                 table[range.charAt(0)] = true;
403             } else if (range.length() == 3 && range.charAt(1) == '-') {
404                 for (int i = range.charAt(0); i <= range.charAt(2); i++) {
405                     table[i] = true;
406                 }
407             }
408         }
409 
410         return table;
411     }

414     /**
415      * Decodes characters of a string that are percent-encoded octets using 
416      * UTF-8 decoding (if needed).
417      * <p>
418      * It is assumed that the string is valid according to an (unspecified) URI 
419      * component type. If a sequence of contiguous percent-encoded octets is 
420      * not a valid UTF-8 character then the octets are replaced with '\uFFFD'.
421      * <p>
422      * If the URI component is of type HOST then any "%" found between "[]" is 
423      * left alone. It is an IPv6 literal with a scope_id.
424      * <p>
425      * If the URI component is of type QUERY_PARAM then any "+" is decoded as
426      * as ' '.
427      * <p>
428      * @param s the string to be decoded.
429      * @param t the URI component type, may be null.
430      * @return the decoded string.
431      * @throws IllegalArgumentException if a malformed percent-encoded octet is
432      *         detected
433      */
434     public static String decode(String s, Type t) {
435         if (s == null) {
436             throw new IllegalArgumentException();
437         }
438 
439         final int n = s.length();
440         if (n == 0) {
441             return s;
442         }
443 
444         // If there are no percent-escaped octets
445         if (s.indexOf('%') < 0) {
446             // If there are no '+' characters for query param
447             if (t == Type.QUERY_PARAM) {
448                 if (s.indexOf('+') < 0) {
449                     return s;
450                 }
451             } else {
452                 return s;
453             }
454         } else {
455             // Malformed percent-escaped octet at the end
456             if (n < 2) // TODO localize
457             {
458                 throw new IllegalArgumentException("Malformed percent-encoded octet at index 1");
459             }
460 
461             // Malformed percent-escaped octet at the end
462             if (s.charAt(n - 2) == '%') // TODO localize
463             {
464                 throw new IllegalArgumentException("Malformed percent-encoded octet at index " + (n - 2));
465             }
466         }
467 
468         if (t == null)
469             return decode(s, n);
470         
471         switch (t) {
472             case HOST :
473                 return decodeHost(s, n);
474             case QUERY_PARAM :
475                 return decodeQueryParam(s, n);
476             default :
477                 return decode(s, n);
478         }
479     }

493     /**
494      * Decode the query component of a URI.
495      * 
496      * @param q the query component in encoded form.
497      * @param decode true of the query parameters of the query component
498      *        should be in decoded form.
499      * @return the multivalued map of query parameters.
500      */
501     public static Map<String, String> decodeQuery(String q, boolean decode) {
502         Map<String, String> queryParameters = new HashMap<String, String>();
503 
504         if (q == null || q.length() == 0) {
505             return queryParameters;
506         }
507 
508         int s = 0, e = 0;
509         do {
510             e = q.indexOf('&', s);
511 
512             if (e == -1) {
513                 decodeQueryParam(queryParameters, q.substring(s), decode);
514             } else if (e > s) {
515                 decodeQueryParam(queryParameters, q.substring(s, e), decode);
516             }
517             s = e + 1;
518         } while (s > 0 && s < q.length());
519 
520         return queryParameters;
521     }

523     private static void decodeQueryParam(Map<String, String> params,
524             String param, boolean decode) {
525         try {
526             int equals = param.indexOf('=');
527             if (equals > 0) {
528                 params.put(
529                         URLDecoder.decode(param.substring(0, equals), "UTF-8"),
530                         (decode) ? URLDecoder.decode(param.substring(equals + 1), "UTF-8") : param.substring(equals + 1));
531             } else if (equals == 0) {
532                 // no key declared, ignore
533             } else if (param.length() > 0) {
534                 params.put(
535                         URLDecoder.decode(param, "UTF-8"),
536                         "");
537             }
538         } catch (UnsupportedEncodingException ex) {
539             // This should never occur
540             throw new IllegalArgumentException(ex);
541         }
542     }

568     /**
569      * Decode the path component of a URI as path segments.
570      *
571      * @param u the URI. If the path component is an absolute path component
572      *        then the leading '/' is ignored and is not considered a delimiator
573      *        of a path segment.
574      * @param decode true if the path segments of the path component
575      *        should be in decoded form.
576      * @return the list of path segments.
577      */
578     public static List<PathSegmentImpl> decodePath(URI u, boolean decode) {
579         String rawPath = u.getRawPath();
580         if (rawPath != null && rawPath.length() > 0 && rawPath.charAt(0) == '/') {
581             rawPath = rawPath.substring(1);
582         }
583         return decodePath(rawPath, decode);
584     }

586     /**
587      * Decode the path component of a URI as path segments.
588      * <p>
589      * Any '/' character in the path is considered to be a deliminator
590      * between two path segments. Thus if the path is '/' then the path segment
591      * list will contain two empty path segments. If the path is "//" then
592      * the path segment list will contain three empty path segments. If the path
593      * is "/a/" the path segment list will consist of the following path
594      * segments in order: "", "a" and "".
595      *
596      * @param path the path component in encoded form.
597      * @param decode true if the path segments of the path component
598      *        should be in decoded form.
599      * @return the list of path segments.
600      */
601     public static List<PathSegmentImpl> decodePath(String path, boolean decode) {
602         List<PathSegmentImpl> segments = new LinkedList<PathSegmentImpl>();
603 
604         if (path == null) {
605             return segments;
606         }
607 
608         int s = 0;
609         int e = -1;
610         do {
611             s = e + 1;
612             e = path.indexOf('/', s);
613 
614             if (e > s) {
615                 decodePathSegment(segments, path.substring(s, e), decode);
616             } else if (e == s) {
617                 segments.add(PathSegmentImpl.EMPTY_PATH_SEGMENT);
618             }
619         } while (e != -1);
620         if (s < path.length()) {
621             decodePathSegment(segments, path.substring(s), decode);
622         } else {
623             segments.add(PathSegmentImpl.EMPTY_PATH_SEGMENT);
624         }
625         return segments;
626     }
627 
628     public static void decodePathSegment(List<PathSegmentImpl> segments, String segment, boolean decode) {
629         int colon = segment.indexOf(';');
630         if (colon != -1) {
631             segments.add(new PathSegmentImpl(
632                     (colon == 0) ? "" : segment.substring(0, colon),
633                     decode,
634                     decodeMatrix(segment, decode)));
635         } else {
636             segments.add(new PathSegmentImpl(
637                     segment,
638                     decode));
639         }
640     }

642     /**
643      * Decode the matrix component of a URI path segment.
644      *
645      * @param pathSegment the path segment component in encoded form.
646      * @param decode true if the matrix parameters of the path segment component
647      *        should be in decoded form.
648      * @return the multivalued map of matrix parameters.
649      */
650     public static Map<String, String> decodeMatrix(String pathSegment, boolean decode) {
651         Map<String, String> matrixMap = new HashMap<String, String>();
652 
653         // Skip over path segment
654         int s = pathSegment.indexOf(';') + 1;
655         if (s == 0 || s == pathSegment.length()) {
656             return matrixMap;
657         }
658 
659         int e = 0;
660         do {
661             e = pathSegment.indexOf(';', s);
662 
663             if (e == -1) {
664                 decodeMatrixParam(matrixMap, pathSegment.substring(s), decode);
665             } else if (e > s) {
666                 decodeMatrixParam(matrixMap, pathSegment.substring(s, e), decode);
667             }
668             s = e + 1;
669         } while (s > 0 && s < pathSegment.length());
670 
671         return matrixMap;
672     }

682             // no key declared, ignore
683         } else if (param.length() > 0) {
684             params.put(
685                     UriComponent.decode(param, UriComponent.Type.MATRIX_PARAM),
686                     "");
687         }
688     }

690     private static String decode(String s, int n) {
691         final StringBuilder sb = new StringBuilder(n);
692         ByteBuffer bb = null;
693 
694         for (int i = 0; i < n;) {
695             final char c = s.charAt(i++);
696             if (c != '%') {
697                 sb.append(c);
698             } else {
699                 bb = decodePercentEncodedOctets(s, i, bb);
700                 i = decodeOctets(i, bb, sb);
701             }
702         }
703 
704         return sb.toString();
705     }

707     private static String decodeQueryParam(String s, int n) {
708         final StringBuilder sb = new StringBuilder(n);
709         ByteBuffer bb = null;
710 
711         for (int i = 0; i < n;) {
712             final char c = s.charAt(i++);
713             if (c != '%') {
714                 if (c != '+')
715                     sb.append(c);
716                 else
717                     sb.append(' ');
718             } else {
719                 bb = decodePercentEncodedOctets(s, i, bb);
720                 i = decodeOctets(i, bb, sb);
721             }
722         }
723 
724         return sb.toString();
725     }

727     private static String decodeHost(String s, int n) {
728         final StringBuilder sb = new StringBuilder(n);
729         ByteBuffer bb = null;
730 
731         boolean betweenBrackets = false;
732         for (int i = 0; i < n;) {
733             final char c = s.charAt(i++);
734             if (c == '[') {
735                 betweenBrackets = true;
736             } else if (betweenBrackets && c == ']') {
737                 betweenBrackets = false;
738             }
739 
740             if (c != '%' || betweenBrackets) {
741                 sb.append(c);
742             } else {
743                 bb = decodePercentEncodedOctets(s, i, bb);
744                 i = decodeOctets(i, bb, sb);
745             }
746         }
747 
748         return sb.toString();
749     }

751     /**
752      * Decode a contigious sequence of percent encoded octets.
753      * <p>
754      * Assumes the index, i, starts that the first hex digit of the first
755      * percent-encoded octet.
756      */
757     private static ByteBuffer decodePercentEncodedOctets(String s, int i, ByteBuffer bb) {
758         if (bb == null)
759             bb = ByteBuffer.allocate(1);
760         else
761             bb.clear();
762 
763         while (true) {
764             // Decode the hex digits
765             bb.put((byte) (decodeHex(s, i++) << 4 | decodeHex(s, i++)));
766 
767             // Finish if at the end of the string
768             if (i == s.length()) {
769                 break;
770             }
771 
772             // Finish if no more percent-encoded octets follow
773             if (s.charAt(i++) != '%') {
774                 break;
775             }
776 
777             // Check if the byte buffer needs to be increased in size
778             if (bb.position() == bb.capacity()) {
779                 bb.flip();
780                 // Create a new byte buffer with the maximum number of possible
781                 // octets, hence resize should only occur once
782                 ByteBuffer bb_new = ByteBuffer.allocate(s.length() / 3);
783                 bb_new.put(bb);
784                 bb = bb_new;
785             }
786         }
787 
788         bb.flip();
789         return bb;
790     }

792     /**
793      * Decodes octets to characters using the UTF-8 decoding and appends
794      * the characters to a StringBuffer.
795      * @return the index to the next unchecked character in the string to decode
796      */
797     private static int decodeOctets(int i, ByteBuffer bb, StringBuilder sb) {
798         // If there is only one octet and is an ASCII character
799         if (bb.limit() == 1 && (bb.get(0) & 0xFF) < 0x80) {
800             // Octet can be appended directly
801             sb.append((char) bb.get(0));
802             return i + 2;
803         } else {
804             // 
805             CharBuffer cb = UTF_8_CHARSET.decode(bb);
806             sb.append(cb.toString());
807             return i + bb.limit() * 3 - 1;
808         }
809     }

811     private static int decodeHex(String s, int i) {
812         final int v = decodeHex(s.charAt(i));
813         if (v == -1) // TODO localize
814         {
815             throw new IllegalArgumentException("Malformed percent-encoded octet at index " + i +
816                     ", invalid hexadecimal digit '" + s.charAt(i) + "'");
817         }
818         return v;
819     }
820     private static final int[] HEX_TABLE = createHexTable();
821 
822     private static int[] createHexTable() {
823         int[] table = new int[0x80];
824         Arrays.fill(table, -1);
825 
826         for (char c = '0'; c <= '9'; c++) {
827             table[c] = c - '0';
828         }
829         for (char c = 'A'; c <= 'F'; c++) {
830             table[c] = c - 'A' + 10;
831         }
832         for (char c = 'a'; c <= 'f'; c++) {
833             table[c] = c - 'a' + 10;
834         }
835         return table;
836     }

838     private static int decodeHex(char c) {
839         return (c < 128) ? HEX_TABLE[c] : -1;
840     }

842     private static boolean isHexCharacter(char c) {
843         return c < 128 && HEX_TABLE[c] != -1;
