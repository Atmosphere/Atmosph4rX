16  package org.atmosphere.atmosph4rx.defaults;
17  
18  import org.atmosphere.atmosph4rx.AxSubscriber;
19  import org.atmosphere.atmosph4rx.AxSubscription;
20  import org.atmosphere.atmosph4rx.AxSocket;
21  import org.atmosphere.atmosph4rx.core.AxMetaData;
22  import org.atmosphere.atmosph4rx.defaults.DefaultAxRouter.AxProcessor;
23  import org.reactivestreams.Processor;
24  import org.reactivestreams.Subscription;
25  
26  class AxSubscriberInvoker<T> implements AxSubscriber<T> {
27  
28      private final AxSubscriber<T> axSubscriber;
29      private final AxMetaData metaData;
30      private final AxProcessor<Processor<String, String>> outputProcessor;
31  
32      public AxSubscriberInvoker(AxSubscriber<T> axSubscriber, AxMetaData metaData, AxProcessor<Processor<String, String>> outputProcessor) {
33          this.axSubscriber = axSubscriber;
34          this.metaData = metaData;
35          this.outputProcessor = outputProcessor;
36      }
37  
38      @Override
39      public void onNext(T payload) {
40          axSubscriber.onNext(payload);
41  
42          AxSocket<Processor<String, String>, String> single = new AxSocket<Processor<String, String>, String>() {
43  
44              @Override
45              public AxSocket<Processor<String, String>, String> publish(String message) {
46                  toProcessor().onNext(message);
47                  return this;
48              }
49  
50              @Override
51              public Processor<String, String> toProcessor() {
52                  return outputProcessor.out();
53              }
54  
55              @Override
56              public String id() {
57                  return outputProcessor.id();
58              }
59  
60              @Override
61              public String toString() {
62                  return "$classname{}";
63              }
64          };
65  
66          axSubscriber.onNext(single);
67          axSubscriber.onNext(single, (String) payload);
68      }
69  
70      @Override
71      public void onSubscribe(Subscription s) {
72          s.request(Integer.MAX_VALUE);
73          axSubscriber.onSubscribe(s);
74  
75          onSubscribe(new AxSubscription(s, metaData, outputProcessor));
76      }
77  
78      @Override
79      public void onComplete() {
80          axSubscriber.onComplete();
81      }
82  
83      @Override
84      public void onError(Throwable t) {
85          axSubscriber.onError(t);
86      }
87  
88      public void onSubscribe(AxSubscription s) {
89          axSubscriber.onSubscribe(s);
90      }
91  
92      public void onNext(byte[] payload) {
93          axSubscriber.onNext(payload);
94      }
95  
96      public <U extends Processor<? super String, ? super String>> void onNext(AxSocket<U, String> single, String payload) {
97          axSubscriber.onNext(single, payload);
98      }
99  
100     @Override
101     public <U extends Processor<? super String, ? super String>, V> void onNext(AxSocket<U, V> single) {
102         axSubscriber.onNext(single);         
103     }
104 
105     public <U extends Processor<? super String, ? super String>> void onNext(AxSocket<U, byte[]> single, byte[] payload) {
106         axSubscriber.onNext(single, payload);
107     }
108 }
